json渲染 ​

使用 JDSV25 > JDS25Tools > ajaxJSON 来实现加载 json 数据

演示 ​

html
<div class="banner swiper-container">
    <div class="swiper-wrapper"></div>
</div>
js
JDS25Tools.ajaxJSON(
    'https://travel.rakuten.co.jp/share/themes/top/config/slide_banner.master.json',
    function(res){
        // console.log(res);
        // 内容渲染
        document.querySelector('.banner.swiper-container .swiper-wrapper').innerHTML = '';
        res.forEach(item => {
            let _html = `
            <div class="swiper-slide">
                <img src="${item.src}" alt="${item.alt}">
            </div>
            `;
            document.querySelector('.banner.swiper-container .swiper-wrapper').innerHTML += _html;
        });
    },
    function(){
        // 回调函数
        new Swiper('.banner.swiper-container', {
            slidesPerView: 3,
            autoplay: 2500
        });
    }
)
